(defun plus (x y)
  (lispif (equal x 0)
           y
           (plus (prec x) (succ y))))

(defun member (x y)
   (lispif (equal y (empty!))
           (empty!)
           (lispif (equal x (car y))
                   y
                   (member x (cdr y)))))

(defun join (x y)
   (lispif (equal x (empty!))
            y
            (lispcons (car x) (join (cdr x) y))))
